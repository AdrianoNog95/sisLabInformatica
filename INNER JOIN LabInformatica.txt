--Inner Join de Usu치rio com Cargo
SELECT U.ID, U.USUARIO, U.SENHA, U.MAT_FUNCIONARIO, U.ID_CARGO, C.NOME_CARGO
FROM USUARIO AS U INNER JOIN CARGO AS C ON U.ID_CARGO = C.ID_CARGO


--inner join de Usu치rio com funcion치rio(Nome Funcion치rio)
SELECT U.ID, U.USUARIO, U.SENHA, U.MAT_FUNCIONARIO, U.ID_CARGO, F.NOME_FUNCIONARIO
FROM USUARIO AS U INNER JOIN FUNCIONARIO AS F ON U.MAT_FUNCIONARIO = F.MAT_FUNCIONARIO


--ComboBox CadastroUsuario
SELECT U.ID, U.USUARIO, U.SENHA, U.MAT_FUNCIONARIO, U.ID_CARGO, F.NOME_FUNCIONARIO, C.NOME_CARGO
FROM USUARIO AS U
INNER JOIN FUNCIONARIO AS F ON U.MAT_FUNCIONARIO = F.MAT_FUNCIONARIO
INNER JOIN CARGO AS C ON U.ID_CARGO = C.ID_CARGO
